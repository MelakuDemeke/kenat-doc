## 🔄 Convert Dates (Gregorian ↔ Ethiopian)

<Tabs
  items={[
    <div key="js" style={{ display: 'flex', alignItems: 'center', gap: '0.4rem' }}><SiJavascript size={16} /> JavaScript</div>,
    <div key="python" style={{ display: 'flex', alignItems: 'center', gap: '0.4rem' }}><SiPython size={16} /> Python</div>
  ]}
>
  <Tabs.Tab>
    ```js
    import { toEC, toGC } from 'kenat';

    const eth = toEC(2025, 6, 2);
    console.log(eth); 
    // → { year: 2017, month: 9, day: 25 }

    const gc = toGC(2017, 9, 25);
    console.log(gc); 
    // → { year: 2025, month: 6, day: 2 }
    ```
  </Tabs.Tab>
  <Tabs.Tab>
    ```python
    from kenat import to_ec, to_gc

    eth = to_ec(2025, 6, 2)
    print(eth)
    # {'year': 2017, 'month': 9, 'day': 25}

    gc = to_gc(2017, 9, 25)
    print(gc)
    # 2025-06-02
    ```
  </Tabs.Tab>
</Tabs>

## 🧮 Date Arithmetic

<Tabs
  items={[
    <div key="js" style={{ display: 'flex', alignItems: 'center', gap: '0.4rem' }}><SiJavascript size={16} /> JavaScript</div>,
    <div key="python" style={{ display: 'flex', alignItems: 'center', gap: '0.4rem' }}><SiPython size={16} /> Python</div>
  ]}
>
  <Tabs.Tab>
    ```js
    const k = new Kenat('2015/6/10');

    k.addDays(5);
    k.addMonths(-1);
    k.addYears(2);

    console.log(k.getEthiopian());
    // → { year: 2017, month: 5, day: 15 }
    ```
  </Tabs.Tab>
  <Tabs.Tab>
    ```python
    from kenat import Kenat

    k = Kenat('2015/6/10')

    k.add_days(5)
    k.add_months(-1)
    k.add_years(2)

    print(k.get_ethiopian())
    # → {'year': 2017, 'month': 5, 'day': 15}
    ```
  </Tabs.Tab>
</Tabs>

## 📏 Difference Between Dates

<Tabs
  items={[
    <div key="js" style={{ display: 'flex', alignItems: 'center', gap: '0.4rem' }}><SiJavascript size={16} /> JavaScript</div>,
    <div key="python" style={{ display: 'flex', alignItems: 'center', gap: '0.4rem' }}><SiPython size={16} /> Python</div>
  ]}
>
  <Tabs.Tab>
    ```js
    const a = new Kenat('2015/5/15');
    const b = new Kenat('2012/5/15');

    console.log(a.diffInDays(b));    // 1095
    console.log(a.diffInMonths(b));  // 39
    console.log(a.diffInYears(b));   // 3
    ```
  </Tabs.Tab>
  <Tabs.Tab>
    ```python
    from kenat import Kenat

    a = Kenat('2015/5/15')
    b = Kenat('2012/5/15')

    print(a.diff_in_days(b))    # 1095
    print(a.diff_in_months(b))  # 39
    print(a.diff_in_years(b))   # 3
    ```
  </Tabs.Tab>
</Tabs>

## 🗓 Generate a Month Grid

<Tabs
  items={[
    <div key="js" style={{ display: 'flex', alignItems: 'center', gap: '0.4rem' }}><SiJavascript size={16} /> JavaScript</div>,
    <div key="python" style={{ display: 'flex', alignItems: 'center', gap: '0.4rem' }}><SiPython size={16} /> Python</div>
  ]}
>
  <Tabs.Tab>
    ```js
    const grid = Kenat.getMonthGrid({ year: 2017, month: 9, useGeez: true });

    console.log(grid.headers);
    // → [ "እሑድ", "ሰኞ", "ማክሰኞ", ... ]

    console.log(grid.days[0]);
    // → { ethiopian: { day: 1 }, gregorian: { day: 9 }, holiday: null }
    ```
  </Tabs.Tab>
  <Tabs.Tab>
    ```python
    from kenat import MonthGrid

    grid = MonthGrid(2017, 9, use_geez=True)

    print(grid.headers)
    # → [ 'እሑድ', 'ሰኞ', 'ማክሰኞ', ... ]

    print(grid.days[0])
    # → {'ethiopian': {'day': 1}, 'gregorian': {'day': 9}, 'holiday': None}
    ```
  </Tabs.Tab>
</Tabs>

## 🕌 Built-in Holidays

<Tabs
  items={[
    <div key="js" style={{ display: 'flex', alignItems: 'center', gap: '0.4rem' }}><SiJavascript size={16} /> JavaScript</div>,
